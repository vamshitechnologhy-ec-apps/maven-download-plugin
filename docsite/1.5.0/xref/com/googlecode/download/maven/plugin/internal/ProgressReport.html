16  package com.googlecode.download.maven.plugin.internal;
17  
18  import java.net.URI;
19  
20  /**
21   * Progress report of a file download operation.
22   */
23  public interface ProgressReport {
24  
25      /**
26       * Triggered to signal initiation of a download operation.
27       *
28       * @param uri the URI of the resource being downloaded
29       * @param total the total length of resource content.
30       */
31      void initiate(URI uri, long total);
32  
33      /**
34       * Triggered to signal successful retrieval of a chunk
35       * of the resource content.
36       *
37       * @param bytesRead the number of bytes retrieved.
38       */
39      void update(long bytesRead);
40  
41      /**
42       * Triggered to signal completion of the download operation.
43       */
44      void completed();
45  
46      /**
47       * Triggered to signal an error occurred during the download operation.
48       */
49      void error(Exception ex);
50  
51  }
